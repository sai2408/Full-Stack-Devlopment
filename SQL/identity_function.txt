IDENTITY(SEED,INCREMENT):-
---------------------------
--> AUTO INCREMENT VALUES ON A COLUMN
--> USED TO GENERATE AUTO INCREMENT VALUES
--> SEED : -
	--STARTING VALUE OF ID
--> INCREMENT : -
	--INCREMENTAL VALUE IN BETWEEN ID'S
==> APPLIED ONLY ON THE NUMERICAL DATA TYPES
	--INT
	--DECIMAL
	--NUMERIC
==> TABLE SHOULD CONTAINS ONLY ONE IDENTITY FUNCTION.(SYSTEM DEFINED)
EXAMPLE : --
	CREATE TABLE EMP(EID INT IDENTITY,ENAME VARCHAR(20),ESAL INT)
	INSERT INTO EMP VALUES(1,'SAI',20000) ----- > NOT ALLOWED
	INSERT INTO EMP VALUES('VARDHAN',40000)
	INSERT INTO EMP(ENAME,ESAL) VALUES('VENKAT',6000)
	INSERT INTO EMP VALUES ('A',1230),('B',12340)

--> WHEN EVER THE IDENTITY_INSERT IS ON THEN ONLY THE USER CAN EXPLITILY ENTER THE VALUES
--> BY DEFAULT THE SEED WILL STARTS FROM 1 AND INCREMENT ONE

==> NOW LET US GIVE OUR SEED AND INCREMENTATION(USER DEFINED)
EXAMPLE : -
	CREATE TABLE EMP1(EID INT IDENTITY(100,5),ENAME VARCHAR(40),ESAL MONEY)
	INSERT INTO EMP1 VALUES('VARDHAN',40000)
	INSERT INTO EMP1(ENAME,ESAL) VALUES('VENKAT',6000)
	INSERT INTO EMP1 VALUES ('A',1230),('B',12340)

==> HOW TO ENTER THE VALUE EXPLICITLY : -
	SYNTAX : -
		SET IDENTITY_INSERT <TN> OGG/ON
		OFF - DEFAULT CONNECTION OF IDENTITY, WE CANNOT INSERT THE VALUE INTO IDENTITY COLUMN
		ON - WE CAN INSERT THE VALUES INTO IDENTITY COLUMN BY EXPLICITLY
	EXAMPLE : -
		SELECT * FROM EMP
		INSERT INTO EMP(EID,ENAME,ESAL) VALUES (4,'AKHIL,15000) -- >NOT ALLOWED
		SET IDENTITY_INSERT EMP ON
		INSERT INTO EMP(EID,ENAME,ESAL) VALUES (4,'AKHIL,15000) -- >ALLOWED
