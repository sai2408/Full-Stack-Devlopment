2.Arithmetic:
-----------------
	--addition(+)
		let x = 100 + 50;
	--subtraction(-)
		let z = x - y;
	--multiplication(8)
		let z = x * y;
	--exponential(**)
		let z = x ** 2; 
		let z = Math.pow(x,2);
	--division(/)
		let z = x / y;
	--moduls(%)
		let z = x % y;

	--decrement(--)
		let x = 5;
